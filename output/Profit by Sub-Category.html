/* Total Profit by Sub-Category */
SELECT p.sub_category 'Sub-Category', SUM(s.profit) 'Total Profit ($)'
FROM sales s
INNER JOIN (
    SELECT DISTINCT product_id, sub_category 
    FROM products
) p ON s.product_id = p.product_id
GROUP BY 1
ORDER BY 2 DESC
