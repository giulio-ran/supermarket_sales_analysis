/* Profit ratio (Sum(profit)/sum(sales)) by State */
select o.state 'State', round(sum(s.profit)/sum(s.sales),2)*100 'Profit Ratio (Profit/Sales) %'
from sales s 
inner join orders o on o.city = s.city
group by 1 order by 2 limit 30
